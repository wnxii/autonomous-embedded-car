add_executable(server_socket
        server_socket.c
        )
target_compile_definitions(server_socket PRIVATE
        #     server_socket_SSID=\"${server_socket_SSID}\"
        #     server_socket_PASSWORD=\"${server_socket_PASSWORD}\"
        NO_SYS=0            # don't want NO_SYS (generally this would be in your lwipopts.h)
        LWIP_SOCKET=1       # we need the socket API (generally this would be in your lwipopts.h)
        )
target_include_directories(server_socket PRIVATE
        ${CMAKE_CURRENT_LIST_DIR}
        ${CMAKE_CURRENT_LIST_DIR}/.. # for our common lwipopts
        )
target_link_libraries(server_socket
        pico_cyw43_arch_lwip_sys_freertos
        pico_stdlib
        pico_lwip_iperf
        FreeRTOS-Kernel-Heap4 # FreeRTOS kernel and dynamic heap
        )
pico_add_extra_outputs(server_socket)
pico_enable_stdio_usb(server_socket 1)
